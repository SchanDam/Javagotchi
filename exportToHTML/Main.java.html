import java.util.Random;
import java.util.Scanner;

public class Main {
    public static void main(String[] args) throws Exception {
        Scanner sc = new Scanner(System.in);
        Random rng = new Random();

        // Character
        int str = 1;
        int def = 1;
        int hp = 100;
        int age = 1;
        int hunger = 3;

        // Einleitung
        System.out.println("Wie ist mein Name?");
        String name = sc.nextLine();
        Thread.sleep(200);
        System.out.println("\nHallo! Ich bin ein Javagotchi (und auch ein Ritter). Ich heiße " + name + " und bin " + age + " Jahr alt. \n\n");
        Thread.sleep(800);
        Ritter();
        Thread.sleep(1500);
        wmdt();

        // Funktionen
        while (true) {
            String input = sc.nextLine();

            if (input.equals("q")) {
                System.out.println("Auf wiedersehen! Bis bald!");
                System.exit(0);
            }
            else if (input.equals("a")) {
                att(str, def, hp, age, hunger);
            }
            else if (input.equals("e")) {
                if (hunger < 10) {
                    hunger ++;
                    System.out.println("Danke für das Essen! Mein Hungerlevel ist: " + hunger);
                }
                else System.out.println("Ich bin satt");
            }
            else if (input.equals("t")) {
                System.out.println("Ich muss meine Fähigkeiten mit Schwert und Schild trainieren!\nEin Jahr vergeht...");
                age++;

                int strIncrease = rng.nextInt(1,3);
                int defIncrease = rng.nextInt(1,2);
                int hpIncrease = rng.nextInt(10,20);
                int hungerDecrease = rng.nextInt(0, 4) -5;

                str += strIncrease;
                def += defIncrease;
                hp += hpIncrease;
                hunger += hungerDecrease;

                Thread.sleep(1000);
                System.out.println("\nIch bin um ein Jahr gealtert. Ich bin nun " + age + " Jahre alt.");
                Thread.sleep(500);
                System.out.println("Ich bin stärker geworden!\n");
                Thread.sleep(200);
                System.out.println("Stärke: +" + strIncrease + " → " + str);
                System.out.println("Verteidigung: +" + defIncrease + " → " + def);
                System.out.println("Lebenspunkte: +" + hpIncrease + " → " + hp);
                System.out.println("Ich bin hungrig! " + hungerDecrease + " → " + hunger + "\n");
                if (hunger < 1) {
                    System.out.println("Ich bin verhungert...");
                    Thread.sleep(100);
                    System.out.println("Das Spiel ist vorbei.");
                    System.exit(0);
                }
                else if (age == 10) {
                    Thread.sleep(200);
                    System.out.println("\nIch bin nun 10 Jahre alt, von nun an bin ich stark genug um mich selbst durchzuschlagen. Vielen Dank für deine Hilfe!");
                    Thread.sleep(200);
                    System.out.println("Du hast das Spiel gewonnen!");
                    System.exit(0);
                }
            }
            else {
                System.out.println("\nungültige Taste");
                Thread.sleep(200);
                wmdt();
            }
        }
    }

    // Character abfragen
    public static void att(int str, int def, int hp, int age, int hunger) {
        System.out.println("\nMeine Character sind:");
        System.out.println("Stärke: " + str);
        System.out.println("Verteidigung: " + def);System.out.println("Lebenspunkte: " + hp);
        System.out.println("Alter: " + age);
        System.out.println("Hungerlevel: " + hunger + "\n");
    }

    // Hauptmenü
    public static void wmdt() throws InterruptedException {
        System.out.println("\n Was möchstest du tun?\n");
        Thread.sleep(500);
        System.out.println("a für Character prüfen");
        Thread.sleep(100);
        System.out.println("e für Essen kochen");
        Thread.sleep(100);
        System.out.println("t für trainieren (ein Jahr wird vergehen. Achtung, Hungerlevel prüfen!)");
        Thread.sleep(100);
        System.out.println("q für beenden\n");
    }

    // Ritter
    public static void Ritter() {
        System.out.println(""" 
                |\\             // 
                 \\\\           _!_ 
                  \\\\         /___\\
                   \\\\        [+++] 
                    \\\\    _ _\\^^^/_ _ 
                     \\\\/ (    '-'  ( ) 
                     /( \\/ | {@}   /\\ \\
                       \\  / \\     / _> ) 
                        "`   >:::;-'`""'-. 
                            /:::/         \\
                           /  /||   {@}   | 
                          (  / (\\         / 
                          / /   \\'-.___.-' 
                        _/ /     \\ \\
                       /___|    /___|""");
    }
}
